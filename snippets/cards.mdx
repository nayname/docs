export const Card = ({ title, href, icon, children, className = "" }) => {
  const body = (
    <div className={"custom-card " + className}>
      {icon && (
        <div className="card-icon">
          <i className={"icon-" + icon}></i>
        </div>
      )}
      <div className="card-content">
        {title && <h3 className="card-title">{title}</h3>}
        <div className="card-description">{children}</div>
      </div>
    </div>
  );
  return href ? <a href={href} className="card-link">{body}</a> : body;
};

export const CardGroup = ({ cols = 3, children, className = "" }) => (
  <div className={"card-grid card-grid-" + cols + " " + className}>
    {children}
  </div>
);
