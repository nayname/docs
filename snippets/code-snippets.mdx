export const CodeExpandable = ({ title, children, icon = "code", language = "solidity" }) => {
  return (
    <Expandable title={title} icon={icon}>
      {children}
    </Expandable>
  )
}

export const LargeCodeBlock = ({ title, code, language = "solidity", filename = "" }) => {
  return (
    <Expandable title={title} icon="file-code">
      {filename && (
        <div className="text-sm text-gray-600 dark:text-gray-400 mb-2 font-mono">
           {filename}
        </div>
      )}
      <div className="relative">
        <CodeGroup>
          <CodeBlock language={language}>
            {code}
          </CodeBlock>
        </CodeGroup>
      </div>
    </Expandable>
  )
}
